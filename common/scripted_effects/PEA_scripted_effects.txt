on_province_trade_building_built_estate_privilege_effect = {
	if = {
		limit = {
			owner = {
				has_estate_privilege = estate_peasants_merchant_labor
			}
		}
		add_province_modifier = {
			name = estate_peasants_merchant_labor_mod
			duration = -1
		}
	}		
}

on_province_trade_building_destroyed_estate_privilege_effect = {
	if = {
		limit = {
			has_province_modifier = estate_peasants_merchant_labor_mod
		}
		remove_province_modifier = estate_peasants_merchant_labor_mod
	}
}

	influence = 0.1

	on_granted = {
		custom_tooltip = estate_peasants_merchant_labor_mod_gt
		tooltip = {
			add_province_modifier = {
				name = estate_peasants_merchant_labor_mod
				duration = -1
			}
		}

	}
	on_revoked = {
		custom_tooltip = estate_peasants_merchant_labor_mod_rt
	}	
	on_granted_province = {
		if = {
			limit = {
				has_trade_building_trigger = yes
			}
			add_province_modifier = {
				name = estate_peasants_merchant_labor_mod
				duration = -1
			}
		}
	}
	on_invalid_province = {
		if = {
			limit = { has_province_modifier = estate_peasants_merchant_labor_mod }
			remove_province_modifier = estate_peasants_merchant_labor_mod
		}
	}
	on_revoked_province = {
		if = {
			limit = { has_province_modifier = estate_peasants_merchant_labor_mod }
			remove_province_modifier = estate_peasants_merchant_labor_mod
		}
	}

	ai_will_do = {
		factor = 0
	}
}

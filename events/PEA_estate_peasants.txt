#Tax exodus
country_event = {
	id = peasants_estate_events.5
	title = peasants_estate_events.5.t
	desc = peasants_estate_events.5.d
	picture = EUROPEAN_REFUGEES_eventPicture 

	trigger = {
		has_estate = estate_peasants
		NOT = {
			estate_loyalty = {
				estate = estate_peasants
				loyalty = 30
			}
		}
		AND = {
			inflation = 10
		}
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_peasants
				modifier = EST_VAL_TAX_EXODUS
			}
		}
	}
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					is_overseas = no
					is_city = yes
					base_tax = 2
				}
				save_event_target_as = event_province_1
			}
			if = {
				limit = {
					has_saved_event_target = event_province_1
				}
				random_owned_province = {
					limit = {
						is_overseas = no
						is_city = yes
						base_tax = 2
						NOT = { province_id = event_target:event_province_1 }
					}
					save_event_target_as = event_province_2
				}
			}
			if = {
				limit = {
					has_saved_event_target = event_province_2
				}
				random_owned_province = {
					limit = {
						is_overseas = no
						is_city = yes
						base_tax = 2
						NOT = { province_id = event_target:event_province_1 }
						NOT = { province_id = event_target:event_province_2 }
					}
					save_event_target_as = event_province_3
				}
			}
			if = {
				limit = {
					has_saved_event_target = event_province_3
				}
				random_owned_province = {
					limit = {
						is_overseas = no
						is_city = yes
						base_tax = 2
						NOT = { province_id = event_target:event_province_1 }
						NOT = { province_id = event_target:event_province_2 }
						NOT = { province_id = event_target:event_province_3 }
					}
					save_event_target_as = event_province_4
				}
			}
			if = {
				limit = {
					has_saved_event_target = event_province_4
				}
				random_owned_province = {
					limit = {
						is_overseas = no
						is_city = yes
						base_tax = 2
						NOT = { province_id = event_target:event_province_1 }
						NOT = { province_id = event_target:event_province_2 }
						NOT = { province_id = event_target:event_province_3 }
						NOT = { province_id = event_target:event_province_4 }
					}
					save_event_target_as = event_province_5
				}
			}
			random_neighbor_country = {
				limit = {
					is_at_war = no
				}
				save_event_target_as = event_country			
			}
		}
	}

	is_triggered_only = yes

	mean_time_to_happen = {
		months = 1
	}

	option = {
		name = peasants_estate_events.5.a #Get this under control!
		event_target:event_province_1 = {
			add_base_tax = -1
		}
		event_target:event_province_2 = {
			add_base_tax = -1
		}
		event_target:event_province_3 = {
			add_base_tax = -1
		}
		event_target:event_province_4 = {
			add_base_tax = -1
		}
		event_target:event_province_5 = {
			add_base_tax = -1
		}
		add_estate_influence_modifier = {
			estate = estate_peasants
			desc = EST_VAL_TAX_EXODUS
			influence = -10
			duration = 5475
		}
		hidden_effect = {
			event_target:event_country = { country_event = { id = peasants_estate_events.8 days = 1 } }
		}
	}
}
